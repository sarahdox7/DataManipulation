Looking at Ratings and SeriesStart date¶

In [371]:
#SeriesStart, IMDb Rating, SarahRating


SeriesStart = %sql SELECT SeriesStart from SHOWS
IMDbRating = %sql SELECT IMDbRating from SHOWS
SarahRating = %sql SELECT SarahRating from SHOWS

x = [SeriesStart]
y1 = [IMDbRating]
y2 = [SarahRating]



plt.figure(figsize = (15,10))
plt.axis([1993,2021,5,10])

plt.scatter(x, y1, color = 'orange',  s=100, marker='^')
plt.scatter(x, y2, color = 'purple',  s=100, marker='o')

plt.title('The Relationship Between Series Start Date and Viewer Ratings', fontsize= 18, fontweight= 'bold', color= 'Black')

plt.xlabel('Series Start Year', fontsize = 14, color = 'black', fontweight = 'bold', labelpad = 20)
plt.ylabel('Rating (1-10)', fontsize = 14, color = 'black', fontweight = 'bold', labelpad = 20)

plt.xticks(fontsize = 15)
plt.yticks(fontsize = 15)

plt.legend(['IMDbRating', 'SarahRating'], loc='lower left', fontsize= 20, bbox_to_anchor=(0.2,0.2), shadow=True, ncol=1)



plt.show()
